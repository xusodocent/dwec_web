<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <script src='https://api.mapbox.com/mapbox-gl-js/v2.6.1/mapbox-gl.js'></script>
    <link href='https://api.mapbox.com/mapbox-gl-js/v2.6.1/mapbox-gl.css' rel='stylesheet' />

    <title>Document</title>
</head>

<body>
    <h1>Exemple de mapa</h1>
    <p>Degut a les limitacions de la API gratuita no centra correctament el mapa quan carrega la pàgina, a mi
        generalment em diu que estic a Kenia. Si feu clic en l'esfera de localitzar si que durà a la ubicació 
        correcta, com podeu vore en la imatge.
    </p>
    <div id='map' style='width: 100%; height: 400px;'></div>
    <script>
        mapboxgl.accessToken = 'pk.eyJ1Ijoiam9ydGkiLCJhIjoiY2xvaWNwMTkyMW9ubjJxbjA5aTg2YmFycyJ9.w0zdUqxA_kjatKYu7sdneA';

        //let map = null;
        document.addEventListener("DOMContentLoaded", async e => {
            navigator.geolocation.getCurrentPosition(pos => {
                console.log();
                console.log(`Latitud: ${pos.coords.latitude}, Longitud: ${pos.coords.longitude}`);
                showMap(pos.coords);
            });
        });

        function showMap(coords) {
            var map = new mapboxgl.Map({
                container: 'map', // ID de l'element HTML que actua com a contenidor
                style: 'mapbox://styles/mapbox/streets-v11', // Estil del mapa
                center: [coords.latitude, coords.longitude], // Coordenades centrals
                //center: [39.0778795, -0.2596927],
                zoom: 9 // Nivell de zoom
            });

            //afegint controls de navegació
            map.addControl(new mapboxgl.NavigationControl());

            //geolocalitzant a l'usuari
            map.addControl(new mapboxgl.NavigationControl());
            map.addControl(new mapboxgl.FullscreenControl());
            map.addControl(new mapboxgl.GeolocateControl({
                positionOptions: {
                    enableHighAccuracy: true
                },
                trackUserLocation: true
            }));
        }


    </script>
</body>

</html>
